## 1. 概述

### 1.1 模块定位

定制协作工作台是 Tailoris 平台的核心业务模块，面向服装纸样定制场景，为客户提供、版师、商家三方协同工作的统一平台。工作台覆盖从需求提交、订单派发、进度跟踪、版本管理、在线沟通到3D预览的完整协作链路。

### 1.2 核心角色

| 角色 | 标识 | 核心职责 | 权限范围 |
|------|------|----------|----------|
| 客户 | customer | 提交定制需求、确认版本、验收交付 | 需求管理、版本确认、评价 |
| 版师 | pattern_maker | 接收订单、制作纸样、提交版本 | 订单接单、版本管理、文件上传 |
| 商家/管理员 | merchant/admin | 订单派发、进度监控、质量审核 | 订单管理、进度监控、审核 |

#### 2.2.3 版师匹配算法

```java
/**
 * 版师匹配服务
 */
@Service
public class PatternMakerMatchService {

    /**
     * 综合评分匹配算法
     */
    public List<PatternMakerScore> matchPatternMakers(Requirement requirement) {
        // 1. 获取符合条件的版师列表
        List<PatternMakerInfo> candidates = findCandidates(requirement);

        // 2. 计算每个版师的综合评分
        List<PatternMakerScore> scores = new ArrayList<>();
        for (PatternMakerInfo maker : candidates) {
            double score = calculateScore(maker, requirement);
            scores.add(new PatternMakerScore(maker, score));
        }

        // 3. 按评分排序，返回Top N
        return scores.stream()
            .sorted(Comparator.comparingDouble(PatternMakerScore::getScore).reversed())
            .limit(10)
            .collect(Collectors.toList());
    }

    /**
     * 综合评分计算
     */
    private double calculateScore(PatternMakerInfo maker, Requirement requirement) {
        double skillScore = calculateSkillScore(maker, requirement) * 0.35;
        double loadScore = calculateLoadScore(maker) * 0.20;
        double ratingScore = maker.getRating() / 5.0 * 100 * 0.25;
        double priceScore = calculatePriceScore(maker, requirement) * 0.20;

        return skillScore + loadScore + ratingScore + priceScore;
    }
}
```

```java
        order.setProjectName(requirement.getGarmentCategory() + "纸样定制");
```

```java
@Service
public class CollabVersionServiceImpl implements CollabVersionService {

    @Override
    public VersionComparisonVO compareVersions(Long versionId1, Long versionId2) {
        CollaborationVersion v1 = versionMapper.selectById(versionId1);
        CollaborationVersion v2 = versionMapper.selectById(versionId2);

        if (v1 == null || v2 == null) {
            throw new BusinessException("版本不存在");
        }
        if (!v1.getCollabOrderId().equals(v2.getCollabOrderId())) {
            throw new BusinessException("只能对比同一订单的版本");
        }

        VersionComparisonVO comparison = new VersionComparisonVO();
        comparison.setVersion1(convertToVO(v1));
        comparison.setVersion2(convertToVO(v2));

        // 文件对比
        comparison.setFileDiffs(compareFiles(v1.getVersionFiles(), v2.getVersionFiles()));

        // 尺寸数据对比
        comparison.setSizeDiffs(compareSizeData(v1, v2));

        // 变更说明Diff
        comparison.setChangeDescriptionDiff = generateDiff(
            v1.getChangeDescription(), v2.getChangeDescription()
        );

        return comparison;
    }

    /**
     * 文件对比：检测新增、删除、修改的文件
     */
    private List<FileDiff> compareFiles(String filesJson1, String filesJson2) {
        List<FileInfo> files1 = JSONUtil.toList(filesJson1, FileInfo.class);
        List<FileInfo> files2 = JSONUtil.toList(filesJson2, FileInfo.class);

        List<FileDiff> diffs = new ArrayList<>();

        // 新增文件
        for (FileInfo f2 : files2) {
            if (files1.stream().noneMatch(f1 -> f1.getFileName().equals(f2.getFileName()))) {
                diffs.add(new FileDiff(f2.getFileName(), FileDiffType.ADDED));
            }
        }

        // 删除文件
        for (FileInfo f1 : files1) {
            if (files2.stream().noneMatch(f2 -> f2.getFileName().equals(f1.getFileName()))) {
                diffs.add(new FileDiff(f1.getFileName(), FileDiffType.DELETED));
            }
        }

        // 修改文件（同名但大小或修改时间不同）
        for (FileInfo f2 : files2) {
            Optional<FileInfo> match = files1.stream()
                .filter(f1 -> f1.getFileName().equals(f2.getFileName()))
                .findFirst();
            if (match.isPresent() && !match.get().equals(f2)) {
                diffs.add(new FileDiff(f2.getFileName(), FileDiffType.MODIFIED));
            }
        }

        return diffs;
    }
}
```

#### 3.3.2 回滚核心逻辑

```java
@Service
public class CollabVersionServiceImpl implements CollabVersionService {

    @Override
    @Transactional(rollbackFor = Exception.class)
    public Long rollbackVersion(RollbackRequest request) {
        // 1. 查询目标版本
        CollaborationVersion targetVersion = versionMapper.selectById(request.getTargetVersionId());
        if (targetVersion == null) {
            throw new BusinessException("目标版本不存在");
        }

        CollaborationOrder order = collabOrderMapper.selectById(targetVersion.getCollabOrderId());

        // 2. 权限校验
        if (!hasRollbackPermission(request.getOperatorId(), order)) {
            throw new BusinessException("无回滚权限");
        }

        // 3. 创建回滚版本（新版本，继承目标版本的文件和描述）
        CollaborationVersion rollbackVersion = new CollaborationVersion();
        rollbackVersion.setCollabOrderId(order.getId());
        rollbackVersion.setVersionNo(generateVersionNo(order.getId()));
        rollbackVersion.setVersionName("回滚至 V" + targetVersion.getVersionNo());
        rollbackVersion.setVersionType(targetVersion.getVersionType());
        rollbackVersion.setCreatorId(request.getOperatorId());
        rollbackVersion.setVersionFiles(targetVersion.getVersionFiles());
        rollbackVersion.setVersionDescription("从版本 V" + targetVersion.getVersionNo() + " 回滚");
        rollbackVersion.setChangeDescription("回滚操作：将当前版本回退至 " + targetVersion.getVersionDescription());
        rollbackVersion.setParentVersionId(request.getTargetVersionId());
        rollbackVersion.setIsLatest(1);
        rollbackVersion.setReviewStatus(0);
        versionMapper.insert(rollbackVersion);

        // 4. 更新之前最新版本的 is_latest 标记
        versionMapper.update(null, new LambdaUpdateWrapper<CollaborationVersion>()
            .set(CollaborationVersion::getIsLatest, 0)
            .eq(CollaborationVersion::getCollabOrderId, order.getId())
            .eq(CollaborationVersion::getIsLatest, 1)
            .ne(CollaborationVersion::getId, rollbackVersion.getId())
        );

        // 5. 记录回滚操作日志
        versionLogMapper.insert(buildRollbackLog(rollbackVersion, targetVersion));

        return rollbackVersion.getId();
    }
}
```

```java
    public List<CollaborationMessage> getOfflineMessages(Long userId) {
        // 1. 从Redis获取最近离线消息
        String redisKey = "collab:offline:msg:" + userId;
        List<Object> cachedMessages = redisTemplate.opsForList().range(redisKey, 0, -1);

        List<CollaborationMessage> messages = new ArrayList<>();

        if (cachedMessages != null && !cachedMessages.isEmpty()) {
            for (Object obj : cachedMessages) {
                messages.add(JSONUtil.toBean(obj.toString(), CollaborationMessage.class));
            }
            // 拉取后清除Redis离线消息
            redisTemplate.delete(redisKey);
        }

        // 2. 从MySQL获取超过7天的离线消息（如果有）
        List<CollaborationMessage> olderMessages = messageMapper.selectList(
            new LambdaQueryWrapper<CollaborationMessage>()
                .eq(CollaborationMessage::getReceiverId, userId)
                .eq(CollaborationMessage::getIsRead, 0)
                .lt(CollaborationMessage::getCreateTime, DateUtil.offsetDay(new Date(), -7))
                .orderByAsc(CollaborationMessage::getCreateTime)
        );
        messages.addAll(olderMessages);

        return messages;
    }

    /**
     * 用户离线时缓存消息到Redis
```

```java
@Service
public class CollabMessageServiceImpl implements CollabMessageService {

    @Override
    @Transactional(rollbackFor = Exception.class)
    public boolean recallMessage(Long messageId, Long userId) {
        // 1. 查询消息
        CollaborationMessage message = messageMapper.selectById(messageId);
        if (message == null) {
            throw new BusinessException("消息不存在");
        }

        // 2. 权限校验：仅发送者可撤回
        if (!message.getSenderId().equals(userId)) {
            throw new BusinessException("无权撤回此消息");
        }

        // 3. 时间校验：2分钟内
        long elapsed = System.currentTimeMillis() - message.getCreateTime().getTime();
        if (elapsed > 2 * 60 * 1000) {
            throw new BusinessException("消息已超过撤回时限");
        }

        // 4. 更新消息状态
        message.setStatus(2);
        message.setRecallTime(new Date());
        messageMapper.updateById(message);

        // 5. 推送撤回通知
        RecallNotification notification = new RecallNotification();
        notification.setMessageId(messageId);
        notification.setCollabOrderId(message.getCollabOrderId());
        notification.setRecallTime(new Date());
        webSocketService.pushRecallNotification(message.getReceiverId(), notification);

        return true;
    }
}
```

```vue
function loadModel(url) {
  if (props.modelType === 'gltf') {
    const loader = new GLTFLoader()
    loader.load(url, (gltf) => {
      if (model) scene.remove(model)
      model = gltf.scene
      scene.add(model)

      // 计算顶点和面数
      let vertices = 0, faces = 0
      model.traverse((child) => {
        if (child.isMesh && child.geometry) {
          vertices += child.geometry.attributes.position.count
          if (child.geometry.index) {
            faces += child.geometry.index.count / 3
          } else {
            faces += child.geometry.attributes.position.count / 3
          }
        }
      })
      vertexCount.value = vertices
      faceCount.value = Math.floor(faces)

      // 自动调整相机位置
      fitCameraToModel(model)
    })
  } else if (props.modelType === 'obj') {
    const loader = new OBJLoader()
    loader.load(url, (obj) => {
      if (model) scene.remove(model)
      model = obj
      scene.add(model)
      fitCameraToModel(model)
    })
  }
}

function fitCameraToModel(object) {
  const box = new THREE.Box3().setFromObject(object)
  const size = box.getSize(new THREE.Vector3())
  const center = box.getCenter(new THREE.Vector3())

  const maxDim = Math.max(size.x, size.y, size.z)
  const fov = camera.fov * (Math.PI / 180)
  let cameraZ = Math.abs(maxDim / 2 / Math.tan(fov / 2))
  cameraZ *= 1.5

  camera.position.set(center.x, center.y + cameraZ * 0.3, center.z + cameraZ)
  controls.target.copy(center)
  controls.update()
}

  requestAnimationFrame(animate)
  controls.update()
  renderer.render(scene, camera)
}

  if (!model) return

    if (child.isMesh && child.material) {
      child.material.wireframe = !child.material.wireframe
    }
  })
}

  if (renderer) renderer.dispose()
  if (controls) controls.dispose()
}
```

```java
@Service
public class OnlineUserService {

    @Autowired
    private RedisTemplate<String, Object> redisTemplate;

    private static final String ONLINE_KEY = "collab:online:";
    private static final String SESSION_KEY = "collab:session:";

    /**
     * 用户上线
     */
    public void userOnline(Long userId, String sessionId) {
        // 记录在线状态
        redisTemplate.opsForValue().set(ONLINE_KEY + userId, sessionId, 2, TimeUnit.HOURS);

        // 记录Session映射
        redisTemplate.opsForValue().set(SESSION_KEY + sessionId, userId, 2, TimeUnit.HOURS);
    }

    /**
     * 用户下线
     */
    public void userOffline(Long userId) {
        redisTemplate.delete(ONLINE_KEY + userId);
    }

    /**
     * 检查用户是否在线
     */
    public boolean isOnline(Long userId) {
        return Boolean.TRUE.equals(redisTemplate.hasKey(ONLINE_KEY + userId));
    }

    /**
     * 获取在线用户列表
     */
    public Set<Long> getOnlineUsers(Long collabOrderId) {
        Set<String> keys = redisTemplate.keys(ONLINE_KEY + "*");
        Set<Long> onlineUsers = new HashSet<>();
        if (keys != null) {
            for (String key : keys) {
                String userIdStr = key.replace(ONLINE_KEY, "");
                onlineUsers.add(Long.parseLong(userIdStr));
            }
        }
        return onlineUsers;
    }

    /**
     * 心跳续约
     */
    public void heartbeat(Long userId) {
        redisTemplate.expire(ONLINE_KEY + userId, 2, TimeUnit.HOURS);
    }
}
```

### 7.5 前端WebSocket连接

```javascript
import SockJS from 'sockjs-client'
import { Client } from '@stomp/stompjs'

class CollabWebSocketManager {
  constructor() {
    this.client = null
    this.reconnectAttempts = 0
    this.maxReconnectAttempts = 10
    this.reconnectDelay = 1000
    this.subscribers = new Map()
  }

  connect(token) {
    return new Promise((resolve, reject) => {
      this.client = new Client({
        webSocketFactory: () => new SockJS('/ws/collab?token=' + token),
        reconnectDelay: this.reconnectDelay,
        heartbeatIncoming: 10000,
        heartbeatOutgoing: 10000,

        onConnect: () => {
          console.log('WebSocket connected')
          this.reconnectAttempts = 0
          resolve()
        },

        onStompError: (frame) => {
          console.error('STOMP error:', frame.headers['message'])
          reject(new Error(frame.headers['message']))
        },

        onWebSocketClose: () => {
          console.warn('WebSocket closed')
          this.handleReconnect()
        },

        onWebSocketError: (event) => {
          console.error('WebSocket error:', event)
        }
      })

      this.client.activate()
    })
  }

  subscribeToMessages(userId, callback) {
    const subscription = this.client.subscribe(
      `/user/${userId}/queue/messages`,
      (message) => callback(JSON.parse(message.body))
    )
    this.subscribers.set('messages', subscription)
  }

  subscribeToProgress(collabOrderId, callback) {
    const subscription = this.client.subscribe(
      `/topic/collab/${collabOrderId}/progress`,
      (message) => callback(JSON.parse(message.body))
    )
    this.subscribers.set('progress', subscription)
  }

  subscribeToVersions(collabOrderId, callback) {
    const subscription = this.client.subscribe(
      `/topic/collab/${collabOrderId}/versions`,
      (message) => callback(JSON.parse(message.body))
    )
    this.subscribers.set('versions', subscription)
  }

  sendMessage(destination, data) {
    if (this.client && this.client.connected) {
      this.client.publish({
        destination: `/app${destination}`,
        body: JSON.stringify(data)
      })
    }
  }

  disconnect() {
    if (this.client) {
      this.client.deactivate()
    }
  }

  handleReconnect() {
    if (this.reconnectAttempts < this.maxReconnectAttempts) {
      this.reconnectAttempts++
      const delay = Math.min(this.reconnectDelay * Math.pow(2, this.reconnectAttempts), 30000)
      console.log(`Reconnecting in ${delay}ms (attempt ${this.reconnectAttempts})`)
      setTimeout(() => this.connect(this.token), delay)
    }
  }
}
```

```java
/**
 * 文件安全检查服务

@Service
public class FileSecurityService {

    /**
     * 文件上传安全检查
     */
    public void validateFile(MultipartFile file) {
        // 1. 文件大小校验
        if (file.getSize() > MAX_FILE_SIZE) {
            throw new BusinessException("文件大小超过限制");
        }

        // 2. 文件类型校验
        String contentType = file.getContentType();
        String extension = getFileExtension(file.getOriginalFilename());
        if (!isAllowedType(contentType, extension)) {
            throw new BusinessException("不支持的文件类型");
        }

        // 3. 文件内容校验（魔数检测）
        if (!validateMagicNumber(file)) {
            throw new BusinessException("文件内容不合法");
        }

        // 4. 病毒扫描（可选）
        if (enableVirusScan) {
            virusScanner.scan(file);
        }

        // 5. XSS内容检测
        if (isTextFile(extension)) {
            String content = readFileContent(file);
            if (containsXSS(content)) {
                throw new BusinessException("文件内容包含不安全代码");
            }
        }
    }
}
```

```java
            log.error("创建协作订单失败, requirementId: {}", request.getRequirementId(), e);
            throw e;
        }
    }
}
```
