【任务：集成 Resend 邮件服务，使用 @tailorbot.top 域名发信】

我们的平台将使用 Resend 作为邮件发送服务。请严格按照以下步骤完成集成，确保发件邮箱为 @tailorbot.top。

第一步：配置环境变量
请在项目根目录的 .env.local 文件中，添加或更新以下变量。务必使用从 Resend 控制台获取的真实值。

# Resend 邮件服务配置
RESEND_API_KEY=你的_Resend_API_Key_在此
EMAIL_FROM=noreply@tailorbot.top # 或 support@tailorbot.top，此域名必须在Resend控制台已验证通过
PLATFORM_NAME=服装纸样平台 # 用于邮件落款

第二步：创建 Resend 邮件服务层
请创建文件：lib/services/email/resend.service.ts
此文件将封装所有通过 Resend API 发送邮件的逻辑。代码如下：

import { Resend } from 'resend';

// 初始化 Resend 客户端
const resend = new Resend(process.env.RESEND_API_KEY);

// 定义邮件发送选项接口
interface SendEmailOptions {
  to: string | string[];
  subject: string;
  html: string;
  text?: string;
  from?: string;
}

// 核心邮件发送函数
export async function sendEmail({
  to,
  subject,
  html,
  text,
  from = process.env.EMAIL_FROM || 'noreply@tailorbot.top',
}: SendEmailOptions): Promise<{ success: boolean; error?: string }> {
  try {
    const { data, error } = await resend.emails.send({
      from,
      to,
      subject,
      html,
      text: text || html.replace(/<[^>]*>/g, ''), // 自动生成纯文本版本
    });

    if (error) {
      console.error('Resend API 错误:', error);
      return { success: false, error: error.message };
    }

    console.log(`邮件发送成功，ID: ${data?.id}`);
    return { success: true };
  } catch (error: any) {
    console.error('发送邮件时出现异常:', error);
    return { success: false, error: error.message };
  }
}

// 示例：发送用户注册欢迎邮件
export async function sendWelcomeEmail(userEmail: string, userName: string) {
  const subject = `欢迎加入${process.env.PLATFORM_NAME || '我们'}！`;
  const html = `
    <div>
      <h2>欢迎，${userName}！</h2>
      <p>感谢您注册我们的服装纸样平台。</p>
      <p>您可以立即开始浏览海量纸样，或与版师开启定制合作。</p>
      <br>
      <p><strong>${process.env.PLATFORM_NAME || '平台团队'}</strong></p>
      <p>发件邮箱：<a href="mailto:${process.env.EMAIL_FROM}">${process.env.EMAIL_FROM}</a></p>
    </div>
  `;

  return await sendEmail({
    to: userEmail,
    subject,
    html,
  });
}
